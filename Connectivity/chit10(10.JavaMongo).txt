# **Step 1: Install MongoDB on Linux**

1. **Update packages:**

```bash
sudo apt update
```

2. **Install MongoDB (Ubuntu example):**

```bash
sudo apt install -y mongodb
```

3. **Start MongoDB service:**

```bash
sudo systemctl start mongodb
sudo systemctl enable mongodb
```

4. **Check if MongoDB is running:**

```bash
sudo systemctl status mongodb
```

5. **Run Mongo Shell to test:**

```bash
mongo
```

* You should see the Mongo shell prompt.

# **Step 5: Java Code for MongoDB Operations**

Create `MongoExample.java` inside `src/main/java/com/institute/`:

```java
package com.institute;

import com.mongodb.client.*;
import org.bson.Document;

import static com.mongodb.client.model.Filters.eq;

public class MongoExample {
    public static void main(String[] args) {
        // 1. Connect to MongoDB
        MongoClient mongoClient = MongoClients.create("mongodb://localhost:27017");

        // 2. Create Database
        MongoDatabase database = mongoClient.getDatabase("InstituteDB");
        System.out.println("Database created: " + database.getName());

        // 3. Create Collection
        MongoCollection<Document> students = database.getCollection("Student");
        System.out.println("Collection: " + students.getNamespace());

        // 4. Insert Document
        Document doc1 = new Document("RollNo", 1)
                .append("Name", "Amit")
                .append("Age", 20)
                .append("Branch", "Computer")
                .append("Address", new Document("City", "Pune").append("State", "MH"))
                .append("Hobbies", new String[]{"Cricket","Reading"});
        students.insertOne(doc1);
        System.out.println("Document inserted");

        // Insert multiple documents
        Document doc2 = new Document("RollNo", 2).append("Name", "Sneha").append("Age", 21)
                .append("Branch", "IT")
                .append("Address", new Document("City", "Mumbai").append("State", "MH"))
                .append("Hobbies", new String[]{"Painting","Swimming"});
        Document doc3 = new Document("RollNo", 3).append("Name", "Rahul").append("Age", 22)
                .append("Branch", "Mechanical")
                .append("Address", new Document("City", "Pune").append("State", "MH"))
                .append("Hobbies", new String[]{"Football"});
        students.insertMany(java.util.Arrays.asList(doc2, doc3));
        System.out.println("Multiple documents inserted");

        // 5. Display Data
        System.out.println("Documents in collection:");
        FindIterable<Document> docs = students.find();
        for (Document d : docs) {
            System.out.println(d.toJson());
        }

        // 6. Update Document
        students.updateOne(eq("RollNo", 2), new Document("$set", new Document("Age", 23)));
        System.out.println("Document updated");

        // 7. Remove Document
        students.deleteOne(eq("RollNo", 3));
        System.out.println("Document deleted");

        // Close connection
        mongoClient.close();
    }
}
```
